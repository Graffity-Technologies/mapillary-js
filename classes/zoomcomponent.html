Class ZoomComponent

classdesc

Component rendering UI elements used for zooming.

example
var viewer = new Mapillary.Viewer(
    "<element-id>",
    "<client-id>",
    "<my key>");

var zoomComponent = viewer.getComponent("zoom");
zoomComponent.configure({ size: Mapillary.ComponentSize.Small });
